import matplotlib.pyplot as plt

def plot_bad_rate_clustered_wide(
    data,
    features,
    feature_col="Feature",
    level_col="Level",
    top_n=20,
    figsize=(14, 5),
    rotation=45
):

    for feature in features:

        temp = data[data[feature_col] == feature].copy()

        if temp.empty:
            continue

        temp = temp.sort_values(
            "All_model_bad_rate",
            ascending=False
        ).head(top_n)

        chart_df = temp.set_index(level_col)[
            [
                "All_model_bad_rate",
                "Funded_model_bad_rate"
            ]
        ]

        ax = chart_df.plot(
            kind="bar",
            figsize=figsize,
            width=0.8
        )

        ax.set_title(f"{feature} Bad Rate Comparison")
        ax.set_xlabel(feature)
        ax.set_ylabel("Bad Rate")

        max_value = chart_df.to_numpy().max()

        for container in ax.containers:
            labels = [
                f"{v:.2%}" if v > max_value * 0.03 else ""
                for v in container.datavalues
            ]

            ax.bar_label(
                container,
                labels=labels,
                padding=3,
                fontsize=8
            )

        plt.xticks(rotation=rotation, ha="right")
        plt.legend(["All Model", "Funded Model"])
        plt.tight_layout()
        plt.show()
