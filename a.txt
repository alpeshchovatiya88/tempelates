def plot_bad_rate_clustered_wide(
    data,
    features,
    feature_col="Feature",
    level_col="Level",
    all_rate_col="All_model_bad_rate",
    funded_rate_col="Funded_model_bad_rate",
    all_count_col="All_model_count",
    funded_count_col="Funded_model_count",
    top_n=20,
    figsize=(20, 8),
    rotation=45,
    rate_is_percent=False
):

    for feature in features:

        temp = data[data[feature_col] == feature].copy()

        if temp.empty:
            continue

        temp = temp.sort_values(
            all_rate_col,
            ascending=False
        ).head(top_n)

        chart_df = temp.set_index(level_col)[
            [
                all_rate_col,
                funded_rate_col
            ]
        ]

        ax = chart_df.plot(
            kind="bar",
            figsize=figsize,
            width=0.7
        )

        ax.set_title(
            f"{feature} - Bad Rate Comparison",
            fontsize=16,
            fontweight="bold"
        )

        ax.set_xlabel(feature, fontsize=12)
        ax.set_ylabel("Bad Rate (%)", fontsize=12)

        max_rate = chart_df.to_numpy().max()
        ax.set_ylim(0, max_rate * 1.35)

        for i, container in enumerate(ax.containers):

            if i == 0:
                counts = temp[all_count_col].values
            else:
                counts = temp[funded_count_col].values

            labels = []

            for rate, count in zip(container.datavalues, counts):

                if rate_is_percent:
                    label = f"{rate:.1f}% | n={count:,.0f}"
                else:
                    label = f"{rate:.1%} | n={count:,.0f}"

                labels.append(label)

            ax.bar_label(
                container,
                labels=labels,
                padding=4,
                fontsize=10
            )

        ax.legend(
            ["All Model", "Funded Model"],
            title="Model",
            fontsize=10
        )

        plt.xticks(
            rotation=rotation,
            ha="right",
            fontsize=10
        )

        plt.yticks(fontsize=10)

        plt.tight_layout()
        plt.show()
