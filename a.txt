import matplotlib.pyplot as plt

def plot_bad_rate_clustered_wide(
    data,
    features,
    feature_col="Feature",
    level_col="Level",
    top_n=20,
    figsize=(20, 8),
    rotation=45,
    rate_is_percent=False
):

    for feature in features:

        temp = data[data[feature_col] == feature].copy()

        if temp.empty:
            continue

        temp = temp.sort_values(
            "All_model_bad_rate",
            ascending=False
        ).head(top_n)

        chart_df = temp.set_index(level_col)[
            [
                "All_model_bad_rate",
                "Funded_model_bad_rate"
            ]
        ]

        ax = chart_df.plot(
            kind="bar",
            figsize=figsize,
            width=0.7
        )

        ax.set_title(
            f"{feature} - Bad Rate Comparison",
            fontsize=16,
            fontweight="bold"
        )

        ax.set_xlabel(feature, fontsize=12)
        ax.set_ylabel("Bad Rate (%)", fontsize=12)

        max_rate = chart_df.to_numpy().max()
        ax.set_ylim(0, max_rate * 1.35)

        for i, container in enumerate(ax.containers):

            # Use BAD COUNT instead of TOTAL COUNT
            if i == 0:
                bad_counts = temp["All_model_bad_count"].values
            else:
                bad_counts = temp["Funded_model_bad_count"].values

            labels = []

            for rate, bad_count in zip(container.datavalues, bad_counts):

                if rate_is_percent:
                    label = f"{rate:.1f}%\nBad={bad_count:,.0f}"
                else:
                    label = f"{rate:.1%}\nBad={bad_count:,.0f}"

                labels.append(label)

            ax.bar_label(
                container,
                labels=labels,
                padding=4,
                fontsize=10
            )

        ax.legend(
            ["All Model", "Funded Model"],
            title="Model"
        )

        plt.xticks(
            rotation=rotation,
            ha="right"
        )

        plt.tight_layout()
        plt.show()
